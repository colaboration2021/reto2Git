CREATE TABLE audience (
    ID NUMBER (10) NOT NULL,
    OWNER VARCHAR2 (20) NULL,
    CAPACITY NUMBER (10) NULL,
    CATEGORY_ID NUMBER (10) NULL,
    NAME VARCHAR2 (400) NULL,

    PRIMARY KEY (ID));

    -- Generate ID using sequence and trigger
CREATE SEQUENCE audience_seq START WITH 1 INCREMENT BY 1;

CREATE OR REPLACE TRIGGER audience_seq_tr
BEFORE INSERT ON audience FOR EACH ROW
WHEN (NEW.ID IS NULL)
BEGIN
    SELECT audience_seq.NEXTVAL INTO : NEW.ID FROM DUAL;
END;
